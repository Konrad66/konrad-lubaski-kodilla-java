SELECT
    ISS.ID AS ISSUE_ID,
    ISS.SUMMARY,
    ISS.DESCRIPTION,
    LISTS.NAME AS LIST_NAME
FROM ISSUES AS ISS
         JOIN ISSUESLISTS AS LISTS ON ISS.ISSUESLIST_ID = LISTS.ID;

SELECT
    ISS.ID AS ISSUE_ID,
    ISS.SUMMARY,
    ISS.DESCRIPTION,
    U.FIRSTNAME,
    U.LASTNAME
FROM ISSUES AS ISS
         LEFT JOIN USERS AS U ON ISS.USER_ID_ASSIGNEDTO = U.ID;

SELECT
    U.FIRSTNAME,
    U.LASTNAME,
    COUNT(ISS.ID) AS TASK_COUNT
FROM USERS AS U
         JOIN ISSUES AS ISS ON U.ID = ISS.USER_ID_ASSIGNEDTO
GROUP BY U.ID, U.FIRSTNAME, U.LASTNAME;
